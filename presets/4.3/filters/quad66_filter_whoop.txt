#$ TITLE: Quad66 4.3 Filter settings for clean build WITHOUT RPM FILTERING.
#$ FIRMWARE_VERSION: 4.3
#$ CATEGORY: FILTERS
#$ STATUS: COMMUNITY
#$ KEYWORDS: quad66, filter
#$ AUTHOR: Quad66
#$ DESCRIPTION: WARNING: This filtering is for situations when you can't or don't want to use RPM filtering. 
#$ DESCRIPTION:
#$ DESCRIPTION: WARNING: May cause fly away on janky props!
#$ DESCRIPTION:
#$ DESCRIPTION: If the quad is too angry try a more conservative D term slider position. 



# -- RPM filtering OFF --
set rpm_filter_harmonics = 0

# -- Gyro filters --
# None
set gyro_lpf1_dyn_min_hz = 0
set gyro_lpf1_dyn_max_hz = 0
set gyro_lpf1_static_hz = 0
set gyro_lpf2_static_hz = 1000

# -- Gyro Sliders (on by default) --
# only lpf2 is active
set simplified_gyro_filter_multiplier = 200

# -- Gyro Static Notches (default)--

# -- Gyro Dynamic Notches --
set dyn_notch_count = 4
set dyn_notch_min_hz = 120
set dyn_notch_max_hz = 900


# -- Dterm filtering --
set dterm_lpf1_dyn_min_hz = 150
set dterm_lpf1_dyn_max_hz = 300
set dterm_lpf1_static_hz = 150
set dterm_lpf2_static_hz = 300

# -- Dterm sliders --
set simplified_dterm_filter_multiplier = 200
