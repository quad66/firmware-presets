#$ TITLE: Quad66 4.3 Filter settings for NOT SO CLEAN WHOOPS with RPM FILTERING.
#$ FIRMWARE_VERSION: 4.3
#$ CATEGORY: FILTERS
#$ STATUS: COMMUNITY
#$ KEYWORDS: quad66, filter
#$ AUTHOR: Quad66
#$ DESCRIPTION: WARNING: This filtering is for slightly noiser whoops and will add some latency. The benefit is a cleaner sounding whoop and better flight time.
#$ DESCRIPTION:
#$ DESCRIPTION: I personally use this filtering on my 0702 whoop build.
#$ DESCRIPTION:
#$ DESCRIPTION: WARNING: Ensure that DShot Telemetry is working properly!
#$ DESCRIPTION:


# -- Gyro filters --
# None
set gyro_lpf1_dyn_min_hz = 0
set gyro_lpf1_dyn_max_hz = 0
set gyro_lpf1_static_hz = 0
set gyro_lpf2_static_hz = 1000

# -- Gyro Sliders (on by default) --
# only lpf2 is active
set simplified_gyro_filter_multiplier = 200

# -- Gyro Static Notches (default)--

# -- Gyro Dynamic Notches --
set dyn_notch_count = 2
set dyn_notch_q = 300
set dyn_notch_min_hz = 100
set dyn_notch_max_hz = 900

# -- RPM filtering --
set dshot_bidir = ON
#set motor_pwm_protocol = DSHOT300
set motor_poles = 12
set rpm_filter_harmonics = 1
set rpm_filter_q = 750

# -- Dterm filtering --
set dterm_lpf1_dyn_min_hz = 112
set dterm_lpf1_dyn_max_hz = 225
set dterm_lpf1_static_hz = 112
set dterm_lpf2_static_hz = 225

# -- Dterm sliders --
set simplified_dterm_filter_multiplier = 150
